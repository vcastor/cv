% Abbreviated author names: "V Castor", "J Guevara", etc.
% \namepartgiveni = "V\bibinitperiod" or "Á\bibinitperiod \bibinitdelim M…"
% Take everything before the first backslash → the single initial letter.
% \namepartfamily = "Castor-Villegas" → take everything before first hyphen.
% Uses string.char(92) for backslash and plain string.find to avoid any
% special-character issues inside \directlua / \DeclareNameFormat.
\DeclareNameFormat{default}{%
  \directlua{
    local gi = '\luaescapestring{\namepartgiveni}'
    local ff = '\luaescapestring{\namepartfamily}'
    local bslash = string.char(92)
    local bpos = string.find(gi, bslash, 1, true)
    local init = bpos and gi:sub(1, bpos - 1) or gi
    local hpos = string.find(ff, '-', 1, true)
    local fam  = hpos and ff:sub(1, hpos - 1) or ff
    local out  = ''
    if string.len(init) > 0 then out = init .. ' ' end
    tex.sprint(out .. fam)
  }%
  \usebibmacro{name:andothers}%
}
